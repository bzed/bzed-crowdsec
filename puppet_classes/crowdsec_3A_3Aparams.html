class crowdsec::params {
  case $facts['kernel'] {
    'Linux' : {
      $config_basedir = '/etc/crowdsec'
      $service_name = 'crowdsec.service'
    }
    default : {
      fail("${facts['kernel']} is not supported by the crowdsec module yet! Please send patches!")
    }
  }
}
